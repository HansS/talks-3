#!/usr/bin/env ruby
require 'mqtt'

MQTT::Client.connect('localhost', 1883) do |c|
  subscription = '8DH1XFGBT2IJX6F8/field1'

  c.get(subscription) do |topic, message|
    api_key, field = topic.split('/')

    feed = Feed.new
    # check api key and other checks
    feed[ field ] = message
